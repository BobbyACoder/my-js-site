  <meta charset="UTF-8">

  <title>JavaScript Example</title>

  <h1 id="greeting">Hello, World!</h1>
  <button onclick="changeGreeting()">Click Me</button>

  <script>
    function changeGreeting() {
      document.getElementById("greeting").innerText = "You clicked the button!";
    }
  </script>
</body>
</html>
